Responsive lightbox

document.addEventListener( "DOMContentLoaded", onLoaded );

function onLoaded (){

var view = new View360();
view.setMode("lightbox-responsive");
view.setConfig({
width:1280,
height:700
});
view.setMode("lightbox-responsive");
view.setFitMode("fit");
view.setImagesDirectory( "./products/mascot" );
view.start();

}
