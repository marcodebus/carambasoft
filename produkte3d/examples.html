In page example

Include js and css files:
<script type="text/javascript" src="./View360.js"></script>
<link rel="stylesheet" type="text/css" href="./View360.css" />
 Code used in this example:
document.addEventListener( "DOMContentLoaded", onLoaded );

function onLoaded (){

var view = new View360();
view.setNavigationConfig({
btnWidth:30,
btnHeight:30
});
view.setImagesPattern( "image_%COL_%ROW.jpg" );
view.setImagesDirectory( "products/truck-1024" );
view.start( document.getElementById("view360") );

}
